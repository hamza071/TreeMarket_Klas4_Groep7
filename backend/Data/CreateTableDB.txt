-- Parent table  

CREATE TABLE Gebruiker ( GebruikerId INT IDENTITY(1,1) PRIMARY KEY, Naam NVARCHAR(255) NOT NULL, Email NVARCHAR(255) NOT NULL, Telefoonnummer NVARCHAR(50) NULL, Wachtwoord NVARCHAR(255) NOT NULL, Rol NVARCHAR(50) NOT NULL, Discriminator NVARCHAR(50) NOT NULL ); 

-- Child tables  

CREATE TABLE Klant ( GebruikerId INT PRIMARY KEY, FOREIGN KEY (GebruikerId) REFERENCES Gebruiker(GebruikerId) ); 

CREATE TABLE Leverancier ( GebruikerId INT PRIMARY KEY, Bedrijf NVARCHAR(255) NOT NULL, KvKNummer NVARCHAR(50) NOT NULL, IBANnummer NVARCHAR(50) NOT NULL, FOREIGN KEY (GebruikerId) REFERENCES Gebruiker(GebruikerId) ); 

CREATE TABLE Veilingsmeester ( GebruikerId INT PRIMARY KEY, PlanDatum DATETIME NULL, FOREIGN KEY (GebruikerId) REFERENCES Gebruiker(GebruikerId) ); 

-- Product table  

CREATE TABLE Product ( ProductId INT IDENTITY(1,1) PRIMARY KEY, Foto NVARCHAR(MAX) NOT NULL, Artikelkenmerken NVARCHAR(MAX) NOT NULL, Hoeveelheid INT NOT NULL, MinimumPrijs DECIMAL(18,2) NOT NULL, Dagdatum DATETIME NOT NULL, LeverancierID INT NOT NULL, FOREIGN KEY (LeverancierID) REFERENCES Leverancier(GebruikerId) ); 

-- Veiling table ] 

CREATE TABLE Veiling ( VeilingID INT IDENTITY(1,1) PRIMARY KEY, Status BIT NULL, StartPrijs DECIMAL(18,2) NOT NULL, EindPrijs DECIMAL(18,2) NOT NULL, HuidigePrijs DECIMAL(18,2) NOT NULL, PrijsStap INT NOT NULL, PrijsStrategie INT NOT NULL, StartTijd DATETIME NOT NULL, EindTijd DATETIME NOT NULL, ProductID INT NOT NULL, VeilingsmeesterID INT NOT NULL, FOREIGN KEY (ProductID) REFERENCES Product(ProductId), FOREIGN KEY (VeilingsmeesterID) REFERENCES Veilingsmeester(GebruikerId) ); 

-- Claim table  

CREATE TABLE Claim ( ClaimID INT IDENTITY(1,1) PRIMARY KEY, Prijs DECIMAL(18,2) NOT NULL, KlantId INT NOT NULL, VeilingId INT NOT NULL, FOREIGN KEY (KlantId) REFERENCES Klant(GebruikerId), FOREIGN KEY (VeilingId) REFERENCES Veiling(VeilingID) ); 

-- Dashboard table  

CREATE TABLE Dashboard ( DashboardID INT IDENTITY(1,1) PRIMARY KEY, ProductID INT NOT NULL, FOREIGN KEY (ProductID) REFERENCES Product(ProductId) );
